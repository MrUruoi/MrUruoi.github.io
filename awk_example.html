[Linux] awk 활용

awk를 이용한 hoge

표계산1 - 각 행의 n열과 m열의 합계

# test.txt 
                    a 0.1 3
                    b -0.65 3
                    a 10.6 -20
                    b 0.22 -30
                
$ cat test.txt | awk '{print $0,$2+$3}'
                    a 0.1 3 3.1
                    b -0.65 3 2.35
                    a 10.6 -20 -9.4
                    b 0.22 -30 -29.78

표계산2 - 각열의 합계

# test.txt 
                    a 0.1 3
                    b -0.65 3
                    a 10.6 -20
                    b 0.22 -30
                
$ cat test.txt | awk '{print;a+=$2;b+=$3} END{print "계",a,b}'
                    a 0.1 3
                    b -0.65 3
                    a 10.6 -20
                    b 0.22 -30
                    계 10.27 -44

표계산3 - 항목별 각열의 합계

# test.txt 
                    a 0.1 3
                    b -0.65 3
                    a 10.6 -20
                    b 0.22 -30
                
$ cat test.txt | awk '{print;x[$1]+=$2;y[$1]+=$3} END{for(k in x){print k "계",x[k],y[k]}}'
                    a 0.1 3
                    b -0.65 3
                    a 10.6 -20
                    b 0.22 -30
                    a계 10.7 -17
                    b계 -0.43 -27
